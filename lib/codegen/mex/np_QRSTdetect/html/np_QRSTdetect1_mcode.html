1function [QRS_amps, QRS_locs, T_amps, T_locs, signal_filtered] = np_QRSTdetect(signal,fs)
  2%==========================================================================
  3% IMPORTANT NOTES:
  4% + For improved accuracy, use this function for limp leads
  5% + This function is inspired by Pan Thompkin algorithm
  6% + Discrete Wavelet Transformation is applied to filter the signal
  7%==========================================================================
  8% Prerequisite files: wavelet_decompose.m
  9%==========================================================================
 10% EKG signal R peak and T peak detection
 11% INPUTS:
 12% + signal    : The given scalar time series ECG
 13% + fs        : Sampling frequency
 14% OUTPUTS
 15% + QRS_amps        : Array of Q wave amplitudes
 16% + QRS_locs        : Array of Q wave location
 17% + T_amps          : Array of T wave amplitudes
 18% + T_locs          : Array of T wave location
 19% + signal_filtered : Signal filtered out baseline and noise
 20%==========================================================================
 21% v 1.0 2017/01/11 22:09:14  Nguyen Pham

 46% Link: https://github.com/nguyenpham95/MY_THESIS

 48signal_filtered = signal;
 49std_thres = 1;
 50sqrt_operation = 4;
 51thres_start = 3;
 52thres_stop = 10;
 53isothres = 10;
 54min_peak_distance = floor(0.4 * fs);
 55%% PREPROCESSING-----------------------------------------------------------   
 56%-CLEAN THE SIGNAL---------------------------------------------------------
 57%-NORMALIZATION CODES------------------------------------------------------
 58signal_filtered = signal_filtered - mean(signal_filtered);
 59L = length(signal_filtered);
 60Ex = 1/L * sum(abs(signal_filtered).^2);
 61signal_filtered = signal_filtered / Ex;
 62%-NORMALIZA THE SIGNAL FROM 0 TO 1
 63if min(signal_filtered) < 0
 64    signal_filtered = signal_filtered + abs(min(signal_filtered));
 65    signal_filtered = signal_filtered / max(signal_filtered);
 66end;
 67%% GENERAL PARAMETERS-------------------------------------------------------
 68QRS_amps = [];
 69QRS_locs = [];
 70T_amps = [];
 71T_locs = [];
 72%-QRS DETECTION-------------------------------------------------         
 73%-READ signal_filtered-------------------------------------------------
 74segment = signal_filtered;
 75%-BRING THE signal_filtered ABOVE 0----------------------------------------
 76if min(segment) < 0
 77   segment = segment + abs(min(segment));
 78end;
 79segment = segment.^sqrt_operation;  
 80thres_mean = (mean(segment)) * ones(1,length(segment));
 81int32 pks;
 82int32 locs;
 83[pks, locs] = findpeaks(segment,'MinPeakDistance',min_peak_distance);
 84for i = 1:length(locs)
 85    if pks(i) > thres_mean(1)
 86       ind = locs(i);
 87       QRS_amps = [QRS_amps signal_filtered(ind)];
 88       QRS_locs = [QRS_locs ind];
 89    end;
 90end;
 91%-T WAVE DETECTION---------------------------------------------------------
 92sig = signal_filtered;
 93%-ZEROLIZING EACH BEAT INTERVAL--------------------------------------------
 94for hk = 1:length(QRS_locs) - 1
 95        data = sig(QRS_locs(hk):QRS_locs(hk + 1));
 96        leng = floor((QRS_locs(hk + 1) - QRS_locs(hk)) / 20);
 97        if leng == 0
 98            leng = 1;
 99        end;
100        isoelec = sig(QRS_locs(hk) + isothres * leng);
101        data = data - isoelec;
102        data = data.^sqrt_operation;
103        data = data(thres_start * leng:thres_stop * leng);
104        [pks, locs] = findpeaks(data);
105        T_threshold = mean(data) + 1 * std(data);
106        if ~isempty(locs)
107            for tp = length(locs):-1:1
108                if pks(tp) > T_threshold
109                    ind = locs(tp);
110                    break;
111                end;
112            end;
113        else
114            ind = floor((thres_stop - thres_start) / 2 * leng);
115        end;
116        if isempty(ind)
117            ind = floor((thres_stop - thres_start) / 2 * leng);
118        end;
119        result1 = QRS_locs(hk) + thres_start * leng + ind;
120        result2 = signal_filtered(QRS_locs(hk) + thres_start * leng + ind);
121        T_locs = [T_locs result1];
122        T_amps = [T_amps result2];
123end;
124%-END OF T DETECTION-------------------------------------------------------
125%-PLOTTING SECTION---------------------------------------------------------
126%figure1 = figure;
127%set(figure1,'name','RT detection','numbertitle','off');
128%plot(signal_filtered);title('signal_filtered');
129%hold on;plot(QRS_locs,QRS_amps,'o');
130%hold on;plot(T_locs,T_amps,'^');
